orders_with_avg_shipping$profit_per_shipping_unit <- NA

for (i in 1:nrow(orders_with_avg_shipping)) {
  if (orders_with_avg_shipping$Shipping.Cost[i] <= orders_with_avg_shipping$avg_shipping_cost[i]) {
    next
  }
  
  orders_with_avg_shipping$profit_per_shipping_unit[i] <- 
    orders_with_avg_shipping$Profit[i] / orders_with_avg_shipping$Shipping.Cost[i]
}
